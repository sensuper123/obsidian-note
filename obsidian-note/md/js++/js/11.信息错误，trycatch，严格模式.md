# SyntaxError 语法错误
```javascript
//Uncaugt SyntaxError : 未捕获的语法错误 === 系统自动抛出的
//Unexpected 意外的 、token 标记  、 Invalid 无效的
var 1 = 1 //Uncaught SyntaxError: Unexpected number
var 1ab = 1 //Uncaught SyntaxError: Invalid or unexpected token
var new = 1 //Uncaught SyntaxError: Unexpected token 'new'
```
# ReferenceError 引用错误
```javascript
//Invalid left-hand side in assignment 左侧赋值无效
console.log(a);//Uncaught ReferenceError: a is not defined
test()//Uncaught ReferenceError: test is not defined
var a = 1 = 2 //Uncaught SyntaxError: Invalid left-hand side in assignment
```
# RangeError 范围错误
```javascript
var arr = [1,2,3]
arr.length = -1 //Uncaught RangeError: Invalid array length
//参数范围错误，只能填 0 - 100
console.log(num.toFixed(-2));//Uncaught RangeError: toFixed() digits argument must be between 0 and 100
```

# URIError 
```javascript
/*
URI 统一资源标致符
URL 统一资源定位符
URN 统一资源名称
URI 包括 URL + URN 等
*/
var url = 'http://www.baidu.com?name=张三'
var newUrl = encodeURI(url)
console.log(newUrl); //http://www.baidu.com?name=%E5%BC%A0%E4%B8%89
var newNewUrl = decodeURI(newUrl)
console.log(newNewUrl);//http://www.baidu.com?name=张三
decodeURI('%s!fas') //Uncaught URIError: URI malformed 
```
# tryCatch throw finally
```javascript
/**
 * tryCatch finally throw
 * try catch 相当于捕获错误，然后执行catch里面的代码，throw可以自定义抛出的错误信息，不影响外部代码
 */
var jsonStr = ''
try {
  if(jsonStr === ''){
    throw 'strJson字符串为空'
  }
  console.log('开始执行');
  var json = JSON.parse(jsonStr)

} catch (error) {
  console.log(error);
  console.log('catch执行');
}finally{
  console.log('finally执行');
}
console.log('外面执行');
//strJson字符串为空
//catch执行
//finally执行
//外面执行
```
